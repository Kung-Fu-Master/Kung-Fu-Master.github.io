# Centos 上更新 Openssl

# 查看centos版本
cat /etc/redhat-release
  CentOS Linux release 7.9.2009 (Core)

# 查看openssl版本
openssl version
  OpenSSL 1.0.2k-fips  26 Jan 2017

# 删除旧版本包
yum remove openssl -y

# 安装依赖
(原始操作)yum install gcc gcc-c++ autoconf automake zlib zlib-devel pcre-devel -y
# 实际安装依赖过程中因为本机已经安装更新过gcc, 因此用了如下命令安装依赖
yum install autoconf automake zlib zlib-devel pcre-devel -y

# 下载包
wget https://www.openssl.org/source/openssl-1.1.1j.tar.gz
tar -zxf openssl-1.1.1j.tar.gz

# 编译
cd openssl-1.1.1j/
./config --prefix=/usr/local/openssl shared zlib
make -j
make install

# 创建软链接
ln -sv /usr/local/openssl/bin/openssl /usr/bin/openssl
ln -sv /usr/local/openssl/include/openssl /usr/include/openssl

# 设置动态库地址
echo '/usr/local/openssl/lib' > /etc/ld.so.conf.d/openssl-ld.conf
ldconfig -v

# 查看版本
openssl version
  OpenSSL 1.1.1j  16 Feb 2021

# 查看某个证书信息
openssl x509 -in public.crt -text -noout

升级完成， 最好重启下服务器.

# debian9 上更新 Openssl

# 当前版本
# openssl version
OpenSSL 1.0.2k-fips  26 Jan 2017

# 删除旧版本包
apt-get --purge remove openssl

# 安装依赖
apt-get update
apt-get install build-essential checkinstall zlib1g-dev -y

# 下载包
wget https://www.openssl.org/source/openssl-1.1.1d.tar.gz
tar -zxf openssl-1.1.1d.tar.gz 

# 编译
cd ./openssl-1.1.1d/
./config --prefix=/usr/local/openssl shared zlib
make
make install

# 创建软链接
ln -sv /usr/local/openssl/bin/openssl /usr/bin/openssl
ln -sv /usr/local/openssl/include/openssl /usr/include/openssl

# 设置动态库地址
echo '/usr/local/openssl/lib' > /etc/ld.so.conf.d/openssl-ld.conf
ldconfig -v

# 查看版本
openssl version
OpenSSL 1.1.1d  10 Sep 2019

升级完成， 最好重启下服务器。
